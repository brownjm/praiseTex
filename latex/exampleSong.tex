\songtitle{Title}
\by{Author}

\chorus{
  \chord{G}This is a \chord{C2}chorus \\
  \chord{Dsus}This is still the \chord{C2}chorus \\
  \chord{G}Another line the awesome \chord{C2}chorus \\
  \chord{Dsus}End of the \chord{C2}chorus
}

\verse {
  \chordleft{Em}This is \chord{C2}verse \\
  \chord{Dsus}This is another line in the verse \\
  \chord{Am}Still another line in the verse \\
  End of the verse
}

\verse {
  This is another verse \\
  Wow, this is an amazing verse \\
  Whew, finally done
}

\bridge {
  \chord{Bm}We have \chord{D}reached the bridge \\
  \chord{G}We have reached the bridge \\
  We have reached the bridge \\
  \chordline{chord{G} \hspace{3mm} \chord{C} \hspace{3mm} \chord{D}}
  \tochorus
}
