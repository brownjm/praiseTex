%    slides.tex - generates presentation slides for praise songs

\documentclass[20pt]{beamer}
\mode<presentation>{\usetheme{default}}

% split up long verses into two slides and allow \framebreak command 
% within \verse{}, see Beamer documentation for proper usage
\setbeamertemplate{frametitle continuation}[from second]

% removes date from title slide
\date{}

% assign title
\newcommand{\songtitle}[1]{\title{#1}}

% assign author and create title slide
\newcommand{\by}[1] {
  \author{#1}
  \maketitle
}

% create a chorus slide
\newcommand{\chorus}[1] {
  \begin{frame}[allowframebreaks] % extra long chorus
    \frametitle{Chorus}
    \begin{center}
      {#1}
    \end{center}
  \end{frame}
}

% create a pre-chorus slide
\newcommand{\prechorus}[1] {
  \begin{frame}
    \frametitle{Pre-Chorus}
    \begin{center}
      {#1}
    \end{center}
  \end{frame}
}

% create a verse slide
\renewcommand{\verse}[1]{
  \begin{frame}[allowframebreaks] % extra long verses
    \frametitle{Verse}
    \begin{center}
      {#1}
    \end{center}
  \end{frame}
}

% create a bridge slide
\newcommand{\bridge}[1]{
  \begin{frame}
    \frametitle{Bridge}
    \begin{center}
      {#1}
    \end{center}
  \end{frame}
}

% allow text to be presented on slides only
\newcommand{\slidesonly}[1]{#1}

% disable unused commands
\newcommand{\chord}[1]{}
\newcommand{\chordleft}[1]{}
\newcommand{\chordline}[1]{}
\renewcommand{\hspace}[1]{}
\newcommand{\tochorus}{}
\newcommand{\toprechorus}{}
\renewcommand{\repeat}{}
\newcommand{\comment}[1]{}
\newcommand{\columnbreak}{}

\end{document}
